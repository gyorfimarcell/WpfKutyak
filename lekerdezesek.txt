SELECT COUNT(*) FROM nevek;

SELECT AVG(életkor) FROM kutyak;

SELECT nevek.kutyanév, fajtak.név FROM kutyak INNER JOIN nevek ON név_id = nevek.id INNER JOIN fajtak ON fajta_id = fajtak.id ORDER BY életkor DESC LIMIT 1;

SELECT fajtak.név, COUNT(*) FROM kutyak INNER JOIN fajtak ON fajta_id = fajtak.id WHERE utolsó_orvosi_ellenõrzés = "2018.01.10" GROUP BY fajtak.név;

SELECT utolsó_orvosi_ellenõrzés, COUNT(*) FROM kutyak GROUP BY utolsó_orvosi_ellenõrzés ORDER BY 2 DESC LIMIT 1;
